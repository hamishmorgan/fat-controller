# M5: Apply Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Implement `config apply` to push config-file changes to Railway with safe confirmation defaults, ordered application, and clear summaries.

**Architecture:** `config apply` in `internal/cli/config_apply.go` will load/merge config files, interpolate `${VAR}` values, fetch live state, compute diffs, and build an ordered apply sequence (settings first, then variables; shared before services; services sorted). A new `internal/apply` package will execute changes through an injected interface, support `--fail-fast`, and return a structured summary. Railway mutations are already available for variables and service settings/limits; apply will adapt `DesiredDeploy` to `ServiceInstanceUpdateInput` and update resource limits via `serviceInstanceLimitsUpdate`.

**Tech Stack:** Go, genqlient (Railway GraphQL), kong, stdlib testing (`testing`, `httptest` for mutation tests)

package apply

// Result holds apply summary counts.
type Result struct {
    Applied int
    Failed  int
    Skipped int
}

// Summary returns a concise summary string.
func (r *Result) Summary() string {
    if r == nil {
        return "applied=0 failed=0 skipped=0"
    }
    return "applied=" + itoa(r.Applied) + " failed=" + itoa(r.Failed) + " skipped=" + itoa(r.Skipped)
}

func itoa(n int) string {
    // Minimal, no fmt for small helper.
    if n == 0 {
        return "0"
    }
    neg := false
    if n < 0 {
        neg = true
        n = -n
    }
    var buf [20]byte
    i := len(buf)
    for n > 0 {
        i--
        buf[i] = byte('0' + n%10)
        n /= 10
    }
    if neg {
        i--
        buf[i] = '-'
    }
    return string(buf[i:])
}
```

package apply

import (
    "context"
    "errors"
    "sort"

    "github.com/hamishmorgan/fat-controller/internal/config"
    "github.com/hamishmorgan/fat-controller/internal/diff"
)

// ErrApplyFailed is a sentinel error used in tests.
var ErrApplyFailed = errors.New("apply failed")

// Options controls apply behavior.
type Options struct {
    FailFast bool
    SkipDeploys bool
}

// Applier executes changes against Railway (service name based).
type Applier interface {
    UpsertVariable(ctx context.Context, service, key, value string, skipDeploys bool) error
    DeleteVariable(ctx context.Context, service, key string) error
    UpdateServiceSettings(ctx context.Context, service string, deploy *config.DesiredDeploy) error
    UpdateServiceResources(ctx context.Context, service string, res *config.DesiredResources) error
}

// Apply computes diffs and executes them in the required order.
func Apply(ctx context.Context, desired *config.DesiredConfig, live *config.LiveConfig, applier Applier, opts Options) (*Result, error) {
    result := &Result{}
    if desired == nil {
        return result, nil
    }
    changes := diff.Compute(desired, live)

    // 1) Settings first (services sorted).
    serviceNames := sortedServiceNames(desired.Services)
    for _, name := range serviceNames {
        sd := changes.Services[name]
        if sd == nil {
            continue
        }
        if hasDeployChanges(sd.Settings) {
            if err := applier.UpdateServiceSettings(ctx, name, desired.Services[name].Deploy); err != nil {
                result.Failed++
                if opts.FailFast {
                    return result, err
                }
            } else {
                result.Applied++
            }
        } else if desired.Services[name].Deploy != nil {
            result.Skipped++
        }

        if hasResourceChanges(sd.Settings) {
            if err := applier.UpdateServiceResources(ctx, name, desired.Services[name].Resources); err != nil {
                result.Failed++
                if opts.FailFast {
                    return result, err
                }
            } else {
                result.Applied++
            }
        } else if desired.Services[name].Resources != nil {
            result.Skipped++
        }
    }

    // 2) Shared variables.
    if changes.Shared != nil {
        for _, ch := range changes.Shared.Variables {
            if err := applyVariable(ctx, applier, "", ch, opts.SkipDeploys); err != nil {
                result.Failed++
                if opts.FailFast {
                    return result, err
                }
            } else {
                result.Applied++
            }
        }
    }

    // 3) Service variables (services sorted).
    for _, name := range serviceNames {
        sd := changes.Services[name]
        if sd == nil {
            continue
        }
        for _, ch := range sd.Variables {
            if err := applyVariable(ctx, applier, name, ch, opts.SkipDeploys); err != nil {
                result.Failed++
                if opts.FailFast {
                    return result, err
                }
            } else {
                result.Applied++
            }
        }
    }

    return result, nil
}

func applyVariable(ctx context.Context, applier Applier, service string, ch diff.Change, skipDeploys bool) error {
    switch ch.Action {
    case diff.ActionCreate, diff.ActionUpdate:
        return applier.UpsertVariable(ctx, service, ch.Key, ch.DesiredValue, skipDeploys)
    case diff.ActionDelete:
        return applier.DeleteVariable(ctx, service, ch.Key)
    default:
        return nil
    }
}

func hasDeployChanges(changes []diff.Change) bool {
    for _, ch := range changes {
        switch ch.Key {
        case "builder", "dockerfile_path", "root_directory", "start_command", "healthcheck_path":
            return true
        }
    }
    return false
}

func hasResourceChanges(changes []diff.Change) bool {
    for _, ch := range changes {
        if ch.Key == "vcpus" || ch.Key == "memory_gb" {
            return true
        }
    }
    return false
}

func sortedServiceNames(services map[string]*config.DesiredService) []string {
    names := make([]string, 0, len(services))
    for name := range services {
        names = append(names, name)
    }
    sort.Strings(names)
    return names
}
```

package apply

import (
    "errors"
    "strings"

    "github.com/hamishmorgan/fat-controller/internal/config"
    "github.com/hamishmorgan/fat-controller/internal/railway"
)

// ToServiceInstanceUpdateInput converts DesiredDeploy to Railway input.
func ToServiceInstanceUpdateInput(desired *config.DesiredDeploy) (railway.ServiceInstanceUpdateInput, error) {
    var input railway.ServiceInstanceUpdateInput
    if desired == nil {
        return input, nil
    }

    if desired.Builder != nil {
        b, ok := parseBuilder(*desired.Builder)
        if !ok {
            return input, errors.New("unknown builder: " + *desired.Builder)
        }
        input.Builder = &b
    }
    if desired.DockerfilePath != nil {
        input.DockerfilePath = desired.DockerfilePath
    }
    if desired.RootDirectory != nil {
        input.RootDirectory = desired.RootDirectory
    }
    if desired.StartCommand != nil {
        input.StartCommand = desired.StartCommand
    }
    if desired.HealthcheckPath != nil {
        input.HealthcheckPath = desired.HealthcheckPath
    }
    return input, nil
}

func parseBuilder(value string) (railway.Builder, bool) {
    switch strings.ToUpper(value) {
    case "NIXPACKS":
        return railway.BuilderNixpacks, true
    case "RAILPACK":
        return railway.BuilderRailpack, true
    case "PAKETO":
        return railway.BuilderPaketo, true
    case "HEROKU":
        return railway.BuilderHeroku, true
    default:
        return "", false
    }
}

package apply

import (
    "context"

    "github.com/hamishmorgan/fat-controller/internal/config"
    "github.com/hamishmorgan/fat-controller/internal/railway"
)

// RailwayApplier applies changes using Railway GraphQL mutations.
type RailwayApplier struct {
    Client        *railway.Client
    ProjectID     string
    EnvironmentID string
    SkipDeploys   bool
}

func (r *RailwayApplier) UpsertVariable(ctx context.Context, service, key, value string, skipDeploys bool) error {
    serviceID, err := railway.ResolveServiceID(ctx, r.Client, r.ProjectID, service)
    if err != nil {
        return err
    }
    return railway.UpsertVariable(ctx, r.Client, r.ProjectID, r.EnvironmentID, serviceID, key, value, skipDeploys)
}

func (r *RailwayApplier) DeleteVariable(ctx context.Context, service, key string) error {
    serviceID, err := railway.ResolveServiceID(ctx, r.Client, r.ProjectID, service)
    if err != nil {
        return err
    }
    return railway.DeleteVariable(ctx, r.Client, r.ProjectID, r.EnvironmentID, serviceID, key)
}

func (r *RailwayApplier) UpdateServiceSettings(ctx context.Context, service string, deploy *config.DesiredDeploy) error {
    if deploy == nil {
        return nil
    }
    serviceID, err := railway.ResolveServiceID(ctx, r.Client, r.ProjectID, service)
    if err != nil {
        return err
    }
    input, err := ToServiceInstanceUpdateInput(deploy)
    if err != nil {
        return err
    }
    return railway.UpdateServiceSettings(ctx, r.Client, serviceID, input)
}

func (r *RailwayApplier) UpdateServiceResources(ctx context.Context, service string, res *config.DesiredResources) error {
    if res == nil {
        return nil
    }
    serviceID, err := railway.ResolveServiceID(ctx, r.Client, r.ProjectID, service)
    if err != nil {
        return err
    }
    return railway.UpdateServiceLimits(ctx, r.Client, r.EnvironmentID, serviceID, res.VCPUs, res.MemoryGB)
}

}

// RunConfigApply is the testable core of `config apply`.
// If desiredOverride is non-nil, it is used instead of loading from disk (testing).
func RunConfigApply(ctx context.Context, globals *Globals, configDir string, extraFiles []string, desiredOverride *config.DesiredConfig, fetcher applyFetcher, applier apply.Applier, out io.Writer) error {
    if out == nil {
        out = os.Stdout
    }

    desired := desiredOverride
    if desired == nil {
        var err error
        desired, err = config.LoadConfigs(configDir, extraFiles)
        if err != nil {
            return err
        }
    }
    if err := config.Interpolate(desired); err != nil {
        return err
    }

    projID, envID, err := fetcher.Resolve(ctx, globals.Workspace, globals.Project, globals.Environment)
    if err != nil {
        return err
    }
    live, err := fetcher.Fetch(ctx, projID, envID, globals.Service)
    if err != nil {
        return err
    }

    // Apply service filter to desired if --service set.
    if globals.Service != "" {
        filtered := &config.DesiredConfig{Shared: desired.Shared, Services: map[string]*config.DesiredService{}}
        if svc, ok := desired.Services[globals.Service]; ok {
            filtered.Services[globals.Service] = svc
        }
        desired = filtered
    }

    // Dry-run by default unless --confirm is set and --dry-run is not.
    if !globals.Confirm || globals.DryRun {
        result := diff.Compute(desired, live)
        formatted := diff.Format(result, globals.ShowSecrets)
        _, err := fmt.Fprintf(out, "dry run: would apply\n%s\n", formatted)
        return err
    }

    // Real apply
    // NOTE: RailwayApplier needs project/environment IDs; set them here when using default applier.
    if ra, ok := applier.(*apply.RailwayApplier); ok {
        ra.ProjectID = projID
        ra.EnvironmentID = envID
        ra.SkipDeploys = globals.SkipDeploys
    }

    res, err := apply.Apply(ctx, desired, live, applier, apply.Options{FailFast: globals.FailFast, SkipDeploys: globals.SkipDeploys})
    if err != nil {
        return err
    }
    _, err = fmt.Fprintln(out, res.Summary())
    return err
}

## Notes / Edge Cases

- If `desired.Services` references a service not in Railway, `ResolveServiceID` will error. That should surface as a failed apply entry and should respect `--fail-fast`.
- Apply should only act on diff changes; if there are no changes, still print a summary (applied=0 failed=0 skipped=0) or a friendly message.
- Shared variables use empty service scope; `ResolveServiceID` already treats empty as shared.
